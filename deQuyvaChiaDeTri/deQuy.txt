double SIN(double)
double COS(double)
double abs(double x)
    return x > 0 ? x : -x;

SIN(double x) 
    if(x < 1e-4) return x;
    return 2 * SIN(x/2) * COS(x/2)

COS(double x) 
    if(abs(x) < 1e -4) return 1;
    double u = SIN(x/2), v = COS(x/2);
    return v^2 - u^2;
